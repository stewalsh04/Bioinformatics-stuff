lm_additive <- lm(light ~ depth + species,
   data = treelight)
lm_additive %>% augment()
# plot the regression lines by species
lm_additive %>%
  augment() %>% 
  ggplot(aes(x = depth, y = .fitted, colour = species)) +
  geom_line() +
  scale_color_brewer(palette = "Dark2")
